        function playSound(type) {
            initAudio();
            if (!audioCtx) return;
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.connect(gain);
            gain.connect(audioCtx.destination);
            const now = audioCtx.currentTime;

            if (type === 'stamp') {
                osc.type = 'triangle'; osc.frequency.setValueAtTime(150, now); osc.frequency.exponentialRampToValueAtTime(40, now + 0.1); gain.gain.setValueAtTime(1, now); gain.gain.exponentialRampToValueAtTime(0.01, now + 0.15); osc.start(now); osc.stop(now + 0.15);

        // --- OYUN DEĞİŞKENLERİ ---
        let questionPool = []; 
        let currentQuestionObj;
        let currentIndexInPool;
        let score = 0;
        let isWaiting = false;

        // DOM Elements
        const paperEl = document.getElementById('paper');
        const qTextEl = document.getElementById('q-text');
        const stampEl = document.getElementById('stamp');
        const sealContainer = document.getElementById('seal-container');
        const scoreEl = document.getElementById('score');
        const remainingCountEl = document.getElementById('remaining-count');
        const modal = document.getElementById('info-modal');
        const modalTitle = document.getElementById('modal-title');
        const modalDesc = document.getElementById('modal-desc');
        const modalStatus = document.getElementById('modal-status');
        const infoCard = document.getElementById('info-card');
        const startScreen = document.getElementById('start-screen');
        const victoryScreen = document.getElementById('victory-screen');

        function initGame() {
            // Havuzu master'dan kopyala
            questionPool = [...masterDatabase];
            score = 0;
            scoreEl.innerText = score;
            startScreen.style.display = 'none';
            victoryScreen.style.display = 'none';
            loadQuestion();
        }

        function loadQuestion() {
            if(questionPool.length === 0) {
                endGame();
                return;
            }

            isWaiting = false;
            remainingCountEl.innerText = questionPool.length;
            
            paperEl.style.animation = 'none';
            paperEl.offsetHeight; 
            paperEl.style.animation = 'slideIn 0.5s ease-out';
            
            stampEl.className = 'stamp-mark';

            // Rastgele soru seç
            currentIndexInPool = Math.floor(Math.random() * questionPool.length);
            currentQuestionObj = questionPool[currentIndexInPool];
            
            qTextEl.innerHTML = currentQuestionObj.q;

            generateSeals();
        }

        function generateSeals() {
            sealContainer.innerHTML = '';
            
            // Doğru cevap + Çeldiriciler
            let options = [currentQuestionObj.a];
            
            if (currentQuestionObj.distractors) {
                options = options.concat(currentQuestionObj.distractors);
            } else {
                while(options.length < 4) {
                    let randomQ = masterDatabase[Math.floor(Math.random() * masterDatabase.length)];
                    if(!options.includes(randomQ.a)) options.push(randomQ.a);
                }
            }

            options.sort(() => Math.random() - 0.5);

            options.forEach(optText => {
                const btn = document.createElement('div');
                btn.className = 'seal-btn';
                btn.innerText = optText;
                btn.onclick = () => handleAnswer(optText, btn);
                sealContainer.appendChild(btn);
            });
        }

        function handleAnswer(selectedAnswer, btnEl) {
            if(isWaiting) return;
            isWaiting = true;

            playSound('stamp');
            btnEl.style.transform = "scale(0.8)";
            setTimeout(() => btnEl.style.transform = "scale(1)", 100);

            const isCorrect = selectedAnswer === currentQuestionObj.a;
            
            if(isCorrect) {
                stampEl.classList.add('approved');
                score += 10;
                // DOĞRU İSE HAVUZDAN SİL
                questionPool.splice(currentIndexInPool, 1);
            } else {
                stampEl.classList.add('rejected');
                // YANLIŞ İSE HAVUZDA KALSIN (Silme yok)
            }
            scoreEl.innerText = score;
            remainingCountEl.innerText = questionPool.length;

            setTimeout(() => {
                showInfoModal(isCorrect);
            }, 600);
        }

        function showInfoModal(isCorrect) {
            playSound(isCorrect ? 'correct' : 'wrong');

            if(isCorrect) {
                modalStatus.innerText = "DOĞRU CEVAP";
                modalStatus.className = "status-badge status-correct";
                infoCard.style.borderLeftColor = "#27ae60";
            } else {
                modalStatus.innerText = "YANLIŞ CEVAP";
                modalStatus.className = "status-badge status-wrong";
                infoCard.style.borderLeftColor = "#c0392b";
            }

            modalTitle.innerText = currentQuestionObj.a; 
            modalDesc.innerText = currentQuestionObj.info;

            modal.style.display = 'flex';
        }

        function closeModal() {
            modal.style.display = 'none';
            paperEl.style.animation = 'slideOut 0.5s forwards';
            setTimeout(loadQuestion, 500);
        }

        function endGame() {
            document.getElementById('final-score').innerText = score;
            victoryScreen.style.display = 'flex';
        }
